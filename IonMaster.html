        // 工具函式

        const getGCD = (a, b) => b ? getGCD(b, a % b) : a;

        const formatSub = text => text.replace(/([0-9]+)/g, '<sub>$1</sub>');


       // 更新題目畫面 (分離邏輯，方便下拉選單切換時直接呼叫)

function renderQuestion() {

    if (!curC || !curA) return;

    const mode = displayModeEl.value;

    

    if (mode === 'symbol') {

        // 加入 <span> 包覆，讓裡面的 sub 和 sup 恢復正常上下標位置

        questionEl.innerHTML = `<span>${curC.html} &nbsp;+&nbsp; ${curA.html}</span>`;

    } else {

        // 這裡也順便包起來保持一致性

        questionEl.innerHTML = `<span>${curC.name}<br>+<br>${curA.name}</span>`;

    }

}


        // 抽取題目事件

        function generateQuestion() {

            curC = cations[Math.floor(Math.random() * cations.length)];

            curA = anions[Math.floor(Math.random() * anions.length)];

            

            answerAreaEl.style.display = 'none'; // 隱藏答案區

            btnAnswer.disabled = false;          // 啟用顯示答案按鈕

            

            renderQuestion();

        }


        // 顯示答案事件

        function showAnswer() {

            if(!curC || !curA) return;


            let cSub = curA.charge; 

            let aSub = curC.charge; 

            let common = getGCD(cSub, aSub);

            cSub /= common; 

            aSub /= common;


            // 名稱處理

            let cleanCName = curC.name.replace(/ion/i, '').trim();

            let cleanAName = curA.name.replace(/ion/i, '').trim();

            document.getElementById('ansName').innerText = `${cleanCName} ${cleanAName.toLowerCase()}`;


            // 陽離子部分處理

            let cPartHtml = formatSub(curC.formula);

            if (cSub > 1) {

                cPartHtml = curC.poly ? `(${cPartHtml})<sub>${cSub}</sub>` : `${cPartHtml}<sub>${cSub}</sub>`;

            }


            // 陰離子部分處理

            let aPartHtml = formatSub(curA.formula);

            if (aSub > 1) {

                aPartHtml = curA.poly ? `(${aPartHtml})<sub>${aSub}</sub>` : `${aPartHtml}<sub>${aSub}</sub>`;

            }


            document.getElementById('ansFormula').innerHTML = cPartHtml + aPartHtml;

            answerAreaEl.style.display = 'block';

            btnAnswer.disabled = true; // 顯示完答案後禁用按鈕，引導按下一題

        }


        // 綁定事件監聽器

        btnAnswer.addEventListener('click', showAnswer);

        displayModeEl.addEventListener('change', renderQuestion); // 切換模式時自動更新當前題目
